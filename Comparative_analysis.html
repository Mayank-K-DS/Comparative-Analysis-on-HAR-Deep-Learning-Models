def train_model(model, criterion, optimizer, num_epochs=10):
    since = time.time()
    train_acc_history, val_acc_history = [], []

    for epoch in range(num_epochs):
        print(f"Epoch {epoch+1}/{num_epochs}")
        print('-'*20)
        
        # Training phase
        model.train()
        running_loss, running_corrects = 0.0, 0
        for inputs, labels in train_loader:
            inputs, labels = inputs.to(device), labels.to(device)
            optimizer.zero_grad()

            outputs = model(inputs)
            _, preds = torch.max(outputs, 1)
            loss = criterion(outputs, labels)

            loss.backward()
            optimizer.step()

            running_loss += loss.item() * inputs.size(0)
            running_corrects += torch.sum(preds == labels.data)

        epoch_loss = running_loss / len(train_dataset)
        epoch_acc = running_corrects.double() / len(train_dataset)
        train_acc_history.append(epoch_acc.item())

        # Validation phase
        model.eval()
        val_corrects = 0
        with torch.no_grad():
            for inputs, labels in val_loader:
                inputs, labels = inputs.to(device), labels.to(device)
                outputs = model(inputs)
                _, preds = torch.max(outputs, 1)
                val_corrects += torch.sum(preds == labels.data)
        val_acc = val_corrects.double() / len(val_dataset)
        val_acc_history.append(val_acc.item())

        print(f"Train Acc: {epoch_acc:.4f} | Val Acc: {val_acc:.4f}")

    time_elapsed = time.time() - since
    print(f"Training complete in {time_elapsed//60:.0f}m {time_elapsed%60:.0f}s")

    return model, train_acc_history, val_acc_history

from sklearn.metrics import classification_report
import numpy as np
import pandas as pd
import matplotlib.pyplot as plt

def evaluate(model):
    model.eval()
    preds, labels_list = [], []
    with torch.no_grad():
        for inputs, labels in val_loader:
            inputs = inputs.to(device)
            outputs = model(inputs)
            _, predictions = torch.max(outputs, 1)
            preds.extend(predictions.cpu().numpy())
            labels_list.extend(labels.numpy())
    return preds, labels_list


results = {}
detailed_reports = {}

for name, info in histories.items():
    print(f"\n===== Evaluating {name} =====")
    preds, labels_list = evaluate(info["model"])
    report = classification_report(labels_list, preds, target_names=train_dataset.classes, output_dict=True, zero_division=0)
    detailed_reports[name] = report

    acc = report["accuracy"]
    macro_f1 = report["macro avg"]["f1-score"]
    weighted_f1 = report["weighted avg"]["f1-score"]
    per_class_f1 = [v["f1-score"] for k, v in report.items() if k in train_dataset.classes]

    results[name] = {
        "accuracy": acc,
        "macro_f1": macro_f1,
        "weighted_f1": weighted_f1,
        "mean_class_f1": np.mean(per_class_f1),
        "std_class_f1": np.std(per_class_f1)
    }

    print(f"Accuracy: {acc:.4f}")
    print(f"Macro F1: {macro_f1:.4f}, Weighted F1: {weighted_f1:.4f}")
    print(f"Mean per-class F1: {np.mean(per_class_f1):.4f}, Std: {np.std(per_class_f1):.4f}")
    print("-" * 70)

from sklearn.metrics import confusion_matrix
import seaborn as sns
import matplotlib.pyplot as plt
import numpy as np
import pandas as pd

# Generate confusion matrices for all models
conf_matrices = {}
class_names = train_dataset.classes

for name, info in histories.items():
    print(f"\nGenerating confusion matrix and insights for {name}...")
    preds, labels_list = evaluate(info["model"])
    cm = confusion_matrix(labels_list, preds)
    conf_matrices[name] = cm

    # --- Plot Confusion Matrix ---
    plt.figure(figsize=(10, 8))
    sns.heatmap(cm, cmap="Blues", cbar=True,
                xticklabels=class_names, yticklabels=class_names)
    plt.title(f"{name} - Confusion Matrix", fontsize=16)
    plt.xlabel("Predicted")
    plt.ylabel("True")
    plt.xticks(rotation=90)
    plt.yticks(rotation=0)
    plt.tight_layout()
    plt.show()

    # --- Compute Per-Class Accuracy ---
    cm_normalized = cm.astype('float') / cm.sum(axis=1)[:, np.newaxis]
    per_class_acc = np.diag(cm_normalized)
    df_per_class = pd.DataFrame({
        "Action": class_names,
        "Accuracy": per_class_acc
    }).sort_values(by="Accuracy", ascending=False)

    print(df_per_class.head(5).to_string(index=False))
    print("\nTop 5 most accurately classified actions for", name)
    print("-" * 60)
    print(df_per_class.head(5))

    print("\nLowest 5 actions (most confused) for", name)
    print("-" * 60)
    print(df_per_class.tail(5))

    # --- Overall Insights ---
    print("\n Insights:")
    print(f"- Mean per-class accuracy: {np.mean(per_class_acc):.3f}")
    print(f"- Std deviation across classes: {np.std(per_class_acc):.3f}")

    # Identify common confusion
    conf_pairs = []
    for i, true_class in enumerate(class_names):
        preds_sorted = np.argsort(cm[i])[::-1]
        if preds_sorted[0] != i:  # most frequent wrong prediction
            conf_pairs.append((true_class, class_names[preds_sorted[0]]))
    print(f"- Example major confusions: {conf_pairs[:5]}")
    print("=" * 80)
